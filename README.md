## Install

```bash
npm install --save react-dashboard-agenda
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'react-dashboard-agenda'
import 'react-dashboard-agenda/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
